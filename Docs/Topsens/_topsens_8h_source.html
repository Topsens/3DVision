278 static int32_t TopsensMapTo2D(const TOPSENS_VECTOR4* projection, const TOPSENS_VECTOR3* position3D, TOPSENS_VECTOR2* position2D)
279 {
280  if (!projection || !position2D || !position3D)
281  {
282  return TOPSENS_ERROR_INVALID_PARAMETER;
283  }
284 
285  position2D->X = position3D->X / position3D->Z * projection->X + projection->Z;
286  position2D->Y = -position3D->Y / position3D->Z * projection->Y + projection->W;
287 
288  return TOPSENS_ERROR_OK;
289 }
